VAR = re.compile(r"\{\{<\s*var\s+([^\s>]+)\s*>\}\}")

def load_yaml(paths):
    data = {}
    for p in paths:
        with open(p, 'r', encoding='utf-8') as f:
            d = yaml.safe_load(f) or {}
        # –ø—Ä–æ—Å—Ç–æ–µ —Å–ª–∏—è–Ω–∏–µ —Å–ª–æ–≤–∞—Ä–µ–π; –ø—Ä–∏ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏ –∑–∞–º–µ–Ω–∏—Ç—å –Ω–∞ deep-merge
        for k, v in d.items():
            if isinstance(v, dict) and isinstance(data.get(k), dict):
                data[k].update(v)
            else:
                data[k] = v
    return data

def get(obj, dotted):
    cur = obj
    for part in dotted.split('.'):
        if isinstance(cur, dict) and part in cur:
            cur = cur[part]
        else:
            return None
    return cur

        'locales/en/docker.yml',
        'locales/en/git.yml',
        'locales/en/kubernetes.yml',

}

# 3) –≤–∞–ª–∏–¥–∞—Ü–∏—è
